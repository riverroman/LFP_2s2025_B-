int a = 10;
int b = 20;
int suma = a + b;
int resta = b - a;
int mult = a * b;
double div = b / 2.0;

System.out.println("Suma: " + suma);
System.out.println("Resta: " + resta);
System.out.println("Multiplicación: " + mult);
System.out.println("División: " + div);

if (suma > 20) {
    System.out.println("Resultado alto");
} else {
    System.out.println("Resultado bajo");
}
